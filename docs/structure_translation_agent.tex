\scnheader{Агент перевода структуры на естественный язык}

\scntext{примечание}{Данный агент переводит переданные ему структуры в текст естественного языка}

\scntext{класс действия}{action\_translate\_structures\_into\_natural\_language}

\begin{scnrelfromvector}{задачи}
  \scnitem{перевести структуры на естественный язык}
\end{scnrelfromvector}

\begin{scnrelfromvector}{аргументы агента}
  \scnitem{structuresSet}
  \begin{scnindent}
    \scntext{пояснение}{structuresSet - sc-множество, содержащее структуры для перевода}
  \end{scnindent}
\end{scnrelfromvector}

\begin{scnrelfromvector}{алгоритм}
  \scnfileitem{Поиск для каждой структуры связей определенного вида и перевод данных связей в строку}
  \scnfileitem{Добавление полученной строки к ответному тексту}
  \scnfileitem{Формирование sc-ссылки с ответным текстом}
\end{scnrelfromvector}
